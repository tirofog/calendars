function createPDF(calendarData) { const { jsPDF } = window.jspdf; const pdf = new jsPDF({ format: 'a4', unit: 'mm', orientation: 'landscape' }); calendarData.forEach(({ month, year, rows }) => { const pageWidth = 297; // A4 Landscape width in mm const rowHeight = 210 / 12; // Row height to evenly fill a page const pages = [ rows.slice(0, 9), rows.slice(9, 21), rows.slice(21), ]; pages.forEach((pageRows, pageIndex) => { if (pdf.internal.getNumberOfPages() > 1 || pageIndex > 0) { pdf.addPage('a4', 'landscape'); } // Draw the vertical month and year label manually pdf.setFontSize(30); const text = `${month} ${year}`; pdf.text(text, 10, 140, { angle: 90 }); // Adjust rotation direction and positioning const startY = pageIndex === 0 ? 210 - (rowHeight * 9) : 0; // Align first page table at the bottom const tableBody = pageRows.map(row => [ row.dayName, row.dayNumber, '', '', '', '', '' ]); pdf.autoTable({ body: tableBody, startY: startY, margin: { top: 0, left: 20, right: 0 }, theme: 'grid', styles: { halign: 'center', valign: 'middle', fontSize: 14, minCellHeight: rowHeight, // Set row height explicitly }, columnStyles: { 0: { cellWidth: 15 }, // Day column 1: { cellWidth: 15 }, // Date column 2: { cellWidth: (pageWidth - 50) / 4 }, // Flexible column 3: { cellWidth: (pageWidth - 50) / 4 }, 4: { cellWidth: (pageWidth - 50) / 4 }, 5: { cellWidth: (pageWidth - 50) / 4 }, }, horizontalPageBreak: false, // Prevent table from breaking across pages didDrawCell: (data) => { if (pageRows[data.row.index]?.isWeekend) { pdf.setFillColor(211, 211, 211); // Light gray color for weekends // Draw the background color first pdf.rect( data.cell.x, data.cell.y, data.cell.width, data.cell.height, 'F' ); pdf.setTextColor(0, 0, 0); // Ensure text is visible pdf.setDrawColor(0, 0, 0); // Ensure borders are visible } }, rowPageBreak: 'auto', tableWidth: 'auto', }); }); }); pdf.save('custom_calendar.pdf'); }
